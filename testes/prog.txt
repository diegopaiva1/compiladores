main() {
  divmod(3, 2)<x, y>;
  print x; -- 1.5
  print y; -- 1.0

  n = 0;

  -- Calculando fat de 0 a 10
  iterate(10) {
    print fat(n)[0];
    n = n + 1;
  }
}

divmod(q :: Int, r :: Int) : Int {
  return q / r, q % r;
}

fat(x :: Int) : Int {
  if (x < 1) {
    return 1;
  }
  else {
    return x * fat(x - 1)[0];
  }
}
