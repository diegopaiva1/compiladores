data Ponto {
  x :: Float;
  y :: Float;
}

dummy(x :: Int, y :: Int) : Int, Int {
  if (x < 3)
    print x;

  if (x < 3) {
    print y;
  }
  else {
    {- Do nothing -}
  }

  iterate (x < 100) {
    x = x + 1;
  }

  print x;

  return x, y;
}

main(args :: Char[][]) {
  read x;
  read y;

  if (((x + y) * 3 == 4))
    print dummy(x, y)[0];
}
