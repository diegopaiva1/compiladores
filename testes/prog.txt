divmod(q :: Int, r :: Int) : Int {
  return q / r, q % r;
}

testIterateFunction(num :: Int) {
  x = 0;

  iterate(num) {
    x = x + 1;
    print x;
  }
}

fat(x :: Int) : Int {
  if (x < 1) {
    return 1;
  }
  else {
    return x * fat(x - 1)[0];
  }
}
